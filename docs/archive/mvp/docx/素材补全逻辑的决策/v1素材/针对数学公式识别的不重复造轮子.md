### 2.  完整代码
```python
import cv2
import numpy as np
import threading
from paddleocr import PaddleOCR

# ====================== 1. 并发安全的OCR引擎单例 ======================
class ThreadSafeMathOCR:
    _instance = None
    _lock = threading.Lock()

    def __new__(cls):
        """单例模式，确保OCR引擎仅初始化一次"""
        if not cls._instance:
            with cls._lock:
                if not cls._instance:
                    cls._instance = super().__new__(cls)
                    # 初始化PaddleOCR，加载公式检测专用配置
                    cls._instance.ocr = PaddleOCR(
                        use_angle_cls=True,       # 支持角度分类（解决倾斜截图）
                        lang="ch",                # 支持中英双语
                        det_db_score_mode="slow", # 提高检测精度
                        use_space_char=True,      # 保留数学符号间的空格
                        show_log=False             # 关闭冗余日志
                    )
        return cls._instance

    def recognize_math_symbols(self, image: np.ndarray, roi_box=None):
        """
        识别截图中的数学符号
        :param image: 输入截图（cv2读取的BGR格式）
        :param roi_box: 候选区坐标 [x1,y1,x2,y2]，None则识别全图
        :return: 识别结果（列表），每个元素：(符号文本, 置信度, 坐标)
        """
        with self._lock:
            # 裁剪候选区（契合方案A：仅识别目标区域）
            if roi_box:
                x1, y1, x2, y2 = roi_box
                image = image[y1:y2, x1:x2]
            
            # 执行OCR识别
            result = self._instance.ocr.ocr(image, cls=True)
            # 解析结果
            math_results = []
            if result and len(result) > 0:
                for line in result[0]:
                    coords = line[0]  # 符号坐标
                    text = line[1][0] # 识别的符号文本
                    score = line[1][1]# 置信度
                    # 过滤非数学符号（可选：保留含公式符号的文本）
                    math_chars = set(r"√∫∑∏+-×÷=<>≤≥≠()[]{}^_/\\")
                    if any(char in math_chars for char in text):
                        math_results.append((text, score, coords))
            return math_results

# ====================== 2. 截图预处理（分辨率自适应+手绘增强） ======================
def preprocess_screenshot(image_path: str, target_width=1920):
    """
    截图预处理：分辨率归一化、手绘边缘增强
    :param image_path: 截图路径
    :param target_width: 基准宽度（1080P）
    :return: 预处理后的图像、分辨率因子res_factor
    """
    # 1. 读取截图
    image = cv2.imread(image_path)
    if image is None:
        raise ValueError(f"无法读取截图：{image_path}")
    
    # 2. 计算分辨率因子（适配480P~4K）
    h, w = image.shape[:2]
    res_factor = target_width / w
    
    # 3. 分辨率归一化（缩放到基准宽度）
    new_w = int(w * res_factor)
    new_h = int(h * res_factor)
    image = cv2.resize(image, (new_w, new_h), interpolation=cv2.INTER_CUBIC)
    
    # 4. 手绘符号增强（降噪+边缘锐化）
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    # 降噪（适配手绘的模糊/噪点）
    gray = cv2.medianBlur(gray, 3)
    # 自适应阈值二值化（增强手绘符号对比度）
    binary = cv2.adaptiveThreshold(
        gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY, 11, 2
    )
    # 边缘增强（让手绘符号轮廓更清晰）
    kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (2,2))
    binary = cv2.morphologyEx(binary, cv2.MORPH_CLOSE, kernel)
    
    return image, binary, res_factor

# ====================== 3. 候选区提取（锁定数学符号区域） ======================
def extract_math_roi(binary_image: np.ndarray, res_factor: float):
    """
    提取数学符号候选区（基于轮廓检测，契合方案A的局部OCR）
    :param binary_image: 二值化后的图像
    :param res_factor: 分辨率因子
    :return: 候选区坐标列表 [[x1,y1,x2,y2], ...]
    """
    # 1. 轮廓检测
    contours, _ = cv2.findContours(
        binary_image, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE
    )
    
    # 2. 过滤极小轮廓（噪声）
    min_area = 50 * (res_factor **2)  # 面积是二维，需平方
    valid_contours = []
    for cnt in contours:
        area = cv2.contourArea(cnt)
        if area >= min_area:
            valid_contours.append(cnt)
    
    # 3. 转换为候选区坐标（向外扩展5px，避免符号截断）
    roi_boxes = []
    expand = int(5 * res_factor)
    for cnt in valid_contours:
        x, y, w, h = cv2.boundingRect(cnt)
        x1 = max(0, x - expand)
        y1 = max(0, y - expand)
        x2 = min(binary_image.shape[1], x + w + expand)
        y2 = min(binary_image.shape[0], y + h + expand)
        roi_boxes.append([x1, y1, x2, y2])
    
    return roi_boxes

# ====================== 4. 主函数：截图识别数学符号 ======================
def recognize_math_symbols_from_screenshot(screenshot_path: str):
    """
    从截图中识别数学符号的主函数
    :param screenshot_path: 截图文件路径
    :return: 识别结果列表，每个元素：(符号文本, 置信度, 候选区坐标)
    """
    # 步骤1：预处理截图
    image, binary, res_factor = preprocess_screenshot(screenshot_path)
    
    # 步骤2：提取数学符号候选区
    roi_boxes = extract_math_roi(binary, res_factor)
    if not roi_boxes:
        print("未检测到数学符号候选区")
        return []
    
    # 步骤3：初始化并发安全的OCR引擎
    ocr_engine = ThreadSafeMathOCR()
    
    # 步骤4：遍历候选区，执行局部OCR
    final_results = []
    for roi in roi_boxes:
        results = ocr_engine.recognize_math_symbols(image, roi)
        for text, score, coords in results:
            final_results.append((text, score, roi))
            print(f"识别到数学符号：{text}，置信度：{score:.2f}，候选区：{roi}")
    
    return final_results

# ====================== 5. 测试用例 ======================
if __name__ == "__main__":
    # 替换为你的截图路径（支持480P~4K，印刷体/手绘公式均可）
    TEST_SCREENSHOT = "math_screenshot.png"
    try:
        results = recognize_math_symbols_from_screenshot(TEST_SCREENSHOT)
        if results:
            print("\n最终识别结果：")
            for idx, (text, score, roi) in enumerate(results, 1):
                print(f"{idx}. 符号：{text}，置信度：{score:.2f}，位置：{roi}")
        else:
            print("未识别到任何数学符号")
    except Exception as e:
        print(f"识别失败：{e}")
```

## 三、 关键适配点说明（契合你之前的核心诉求）
### 1.  分辨率自适应
- 所有像素级参数（轮廓最小面积、候选区扩展像素）都乘以`res_factor`，适配480P~4K截图；
- 预处理阶段将截图归一化到1920px宽度，消除分辨率差异对识别的影响。

### 2.  手绘符号增强
- 用`cv2.medianBlur`降噪，解决手绘截图的噪点问题；
- 自适应阈值二值化（`cv2.adaptiveThreshold`），适配不同亮度的手绘符号；
- 形态学闭运算（`cv2.MORPH_CLOSE`），填补手绘符号的笔画断点。

### 3.  局部OCR（方案A）
- 先通过轮廓检测锁定数学符号候选区，仅对这些小区域执行OCR，避免全图OCR的性能损耗；
- 候选区向外扩展5px，防止手绘符号边缘被截断导致识别失败。

### 4.  并发安全
- OCR引擎封装为带`Lock`的单例，避免多进程/线程调用时的初始化崩溃（解决你之前的`AttributeError`问题）；
- 所有OCR调用都通过单例的带锁方法执行，杜绝资源竞争。

## 四、 效果优化建议（针对手绘/低质量截图）
1.  **添加微型模板库校验**：如果OCR识别结果置信度＜0.7，用OpenCV的`cv2.matchShapes`匹配根号、积分号等核心符号的模板，补充验证；
    ```python
    # 示例：校验根号符号
    def verify_sqrt(contour, res_factor):
        template = cv2.imread("sqrt_template.png", 0)
        template = cv2.resize(template, None, fx=res_factor, fy=res_factor)
        template_cnt, _ = cv2.findContours(template, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        score = cv2.matchShapes(contour, template_cnt[0], cv2.CONTOURS_MATCH_I2, 0.0)
        return score < 0.4  # 得分越低，匹配度越高
    ```
2.  **过滤误识别**：维护数学符号白名单（`√∫∑∏+-×÷=<>≤≥≠()[]{}^_/\\`），仅保留包含这些符号的识别结果；
3.  **LaTeX格式转换**：将识别到的符号转为LaTeX格式（如`^2`→`$^2$`，`√`→`$\sqrt{}$`），直接用于结构化笔记。
