要实现手机端点击词语自动选择名词及其修饰词、添加下划线并获取选中内容，需结合**文本分词标签化**、**NLP语义分析**和**前端交互**。以下是完整实现方案（含代码示例）：


### 一、核心思路
1. **文本预处理**：将文本分词并包裹可点击标签（如`<span>`）。
2. **点击事件捕获**：监听点击，获取被点击词语及上下文。
3. **NLP分析**：通过分词+词性标注/依存句法分析，识别名词及其修饰词。
4. **选中与样式**：给目标词语添加下划线样式，提取选中内容。


### 二、技术选型
- **中文分词**：`jieba.js`（轻量级前端分词库，支持词性标注）。
- **NLP分析**：基于词性规则提取名词短语（或调用百度/腾讯NLP API做依存句法分析，更精准）。
- **前端交互**：原生JS处理点击、样式操作。


### 三、完整代码示例
```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0, user-scalable=no">
    <title>点击选词示例</title>
    <!-- 引入jieba.js（中文分词+词性标注） -->
    <script src="https://cdn.jsdelivr.net/npm/jieba-js@1.0.0/jieba.js"></script>
    <style>
        /* 词语可点击样式 */
        .word {
            cursor: pointer;
            margin: 0 2px;
            line-height: 1.8;
            font-size: 16px;
        }
        /* 选中后下划线样式 */
        .selected {
            text-decoration: underline;
            text-decoration-color: #007AFF;
            text-decoration-thickness: 2px;
        }
        #result {
            margin-top: 20px;
            padding: 10px;
            background: #f5f5f5;
            border-radius: 8px;
        }
    </style>
</head>
<body>
    <div id="text-container"></div>
    <div id="result">选中的词：</div>

    <script>
        // 示例文本
        const rawText = "我有一个红色的苹果，它非常甜。今天天气很好，我想去美丽的公园散步。";
        
        // 1. 初始化jieba并分词+词性标注
        jieba.load().then(() => {
            // 分词+词性标注（结果格式：[{word: "红色", tag: "a"}, {word: "苹果", tag: "n"}, ...]）
            const taggedWords = jieba.tag(rawText);
            renderText(taggedWords); // 渲染文本
        });

        // 2. 渲染文本：将每个词语包裹<span>
        function renderText(taggedWords) {
            const container = document.getElementById('text-container');
            taggedWords.forEach((item, index) => {
                const span = document.createElement('span');
                span.className = 'word';
                span.textContent = item.word;
                span.dataset.index = index; // 存储索引，方便后续查找
                span.dataset.tag = item.tag; // 存储词性
                container.appendChild(span);
            });
            bindClickEvent(taggedWords); // 绑定点击事件
        }

        // 3. 绑定点击事件
        function bindClickEvent(taggedWords) {
            const container = document.getElementById('text-container');
            container.addEventListener('click', (e) => {
                if (!e.target.classList.contains('word')) return;

                // 清除之前的选中状态
                document.querySelectorAll('.selected').forEach(el => el.classList.remove('selected'));

                const clickedIndex = parseInt(e.target.dataset.index);
                const clickedTag = e.target.dataset.tag;

                // 4. 提取名词短语（名词+修饰词）
                const selectedPhraseIndices = extractNounPhrase(taggedWords, clickedIndex);
                
                // 5. 给选中的词语添加下划线
                const allWords = document.querySelectorAll('.word');
                let selectedText = '';
                selectedPhraseIndices.forEach(idx => {
                    allWords[idx].classList.add('selected');
                    selectedText += allWords[idx].textContent;
                });

                // 6. 显示选中的词
                document.getElementById('result').textContent = `选中的词：${selectedText}`;
            });
        }

        // 4. 核心逻辑：基于词性规则提取名词短语（修饰词+名词）
        function extractNounPhrase(taggedWords, clickedIndex) {
            const result = [clickedIndex];
            const clickedWord = taggedWords[clickedIndex];

            // 定义词性规则：
            // - 修饰词：形容词(a)、区别词(b)、代词(r)、数词(m)、量词(q)、状态词(z)
            // - 名词：名词(n)、人名(nr)、地名(ns)、机构名(nt)、专名(nz)
            const modifierTags = ['a', 'b', 'r', 'm', 'q', 'z'];
            const nounTags = ['n', 'nr', 'ns', 'nt', 'nz'];

            // 情况1：点击的是名词 -> 向前找修饰词
            if (nounTags.includes(clickedWord.tag)) {
                let i = clickedIndex - 1;
                while (i >= 0 && modifierTags.includes(taggedWords[i].tag)) {
                    result.unshift(i); // 向前添加修饰词
                    i--;
                }
            }

            // 情况2：点击的是修饰词 -> 向后找名词，再向前找更多修饰词
            else if (modifierTags.includes(clickedWord.tag)) {
                // 先向后找名词
                let j = clickedIndex + 1;
                let foundNoun = false;
                while (j < taggedWords.length) {
                    if (nounTags.includes(taggedWords[j].tag)) {
                        result.push(j);
                        foundNoun = true;
                        break;
                    }
                    j++;
                }
                // 找到名词后，再向前找更多修饰词
                if (foundNoun) {
                    let i = clickedIndex - 1;
                    while (i >= 0 && modifierTags.includes(taggedWords[i].tag)) {
                        result.unshift(i);
                        i--;
                    }
                }
            }

            return result.sort((a, b) => a - b); // 按索引排序
        }
    </script>
</body>
</html>
```


### 四、关键说明
1. **中文分词与词性标注**：  
   使用`jieba.js`实现前端分词，词性标签说明（部分）：
   - 名词：`n`（普通名词）、`nr`（人名）、`ns`（地名）。
   - 修饰词：`a`（形容词）、`m`（数词）、`q`（量词）。

2. **名词短语提取规则**：  
   代码中通过**修饰词+名词**的词性组合提取短语，若需更精准的修饰关系（如“的”字结构、定语从句），建议：
   - 调用**百度NLP依存句法分析API**（免费额度充足），可直接获取词语间的修饰关系（如“红色”修饰“苹果”）。
   - 示例：百度NLP返回的依存关系中，若“红色”的父节点是“苹果”，则两者构成名词短语。

3. **手机端适配**：  
   - 添加`<meta name="viewport" content="width=device-width, initial-scale=1.0, user-scalable=no">`禁止缩放，避免点击偏移。
   - 若需消除手机端点击300ms延迟，可使用`touchstart`事件替代`click`，或引入`fastclick`库。
